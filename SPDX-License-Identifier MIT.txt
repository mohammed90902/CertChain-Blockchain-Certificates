pragma solidity ^0.8.0;

contract CertificateVerifier {
    struct Certificate {
        string studentName;
        string courseName;
        string grade;
    }

    mapping(uint256 => Certificate) public certificates;

    function addCertificate(uint256 _id, string memory _name, string memory _course, string memory _grade) public {
        certificates[_id] = Certificate(_name, _course, _grade);
    }

    function getCertificate(uint256 _id) public view returns (string memory, string memory, string memory) {
        Certificate memory cert = certificates[_id];
        return (cert.studentName, cert.courseName, cert.grade);
    }
}
